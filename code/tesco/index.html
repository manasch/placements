Coding Questions#

  1. Don’t print rows containing -1: Given a square 2D array of integers, don’t print rows that contain -1
  1. Index at which it becomes greater: Given N and P, form a circle of all the primes under N (including) and start from 2 and keep adding values in clockwise manner till the sum becomes greater than P, return the index at which the sum crosses P.
void generatePrimes(int n, vector<int>& primes) {
    primes.clear();
    vector<bool> isPrime(n + 1, true);

    for (int i = 2; i * i <= n; ++i) {
        if (isPrime[i]) {
            for (int j = i * i; j <= n; j += i) {
                isPrime[j] = false;
            }
        }
    }

    for (int i = 2; i <= n; ++i) {
        if (isPrime[i]) {
            primes.push_back(i);
        }
    }
}

int solve(int n, int p) {
    vector<int> primes;
    generatePrimes(n, primes);

    int idx = 0;
    int s = 0;
    while (s > p) {
        s += primes[idx % primes.size()];
        ++idx;
    }
    return (idx - 1) % primes.size();
}
